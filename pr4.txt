#include <iostream>
using namespace std;
class Rectangle
{
    public:
    int length,width;
    Rectangle(int l, int w)
    {
        length=l;
        width=w;
    }
    ~Rectangle()
    {
        cout<<"Rectangle object destroyed"<<endl;
    }
    void area()
    {
        int res;
        res=length*width;
        cout<<"Area of Rectangle="<<res<<endl;
    }
    void perimeter()
    {
        int res;
        res=2*(length+width);
        cout<<"Perimeter of Rectangle="<<res<<endl;
    }
};
int main()
{
    int l,w;
    cout<<"enter value for length:";
    cin>>l;
    cout<<"enter value for width:";
    cin>>w;
    Rectangle r(l,w);
    r.area();
    r.perimeter();
    return 0;
}

Output:
enter value for length:20
enter value for width:15
Area of Rectangle=300
Perimeter of Rectangle=70
Rectangle object destroyed
