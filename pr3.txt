#include <iostream>
using namespace std;
class SavingAccount
{
    public:
    int accno;
    string accname;
    float balance;
    int interestrate;
    SavingAccount(int no, string name,float bal, int irate)
    {
        accno=no;
        accname=name;
        bal=balance;
        interestrate=irate;
    }
    void deposit(double amount)
    {
        if(amount>0)
        {
            balance+=amount;
            cout<<"Deposited:"<<amount<<endl;
        }
    }
    void withdraw(double amount)
    {
        if(amount>0 && amount<=balance)
        {
            balance-=amount;
            cout<<"withdrawn:"<<amount<<endl;
        }
        else
        {
            cout<<"insufficient balance!"<<endl;
        }
    }
    void applyInterest()
    {
        double interest=balance*interestrate/100;
        balance+=interest;
        cout<<"interest applied:"<<interest<<endl;
    }
    void display()
    {
        cout<<"\n[SAVING ACCOUNT]"<<endl;
        cout<<"Account Holder="<<accname<<endl;
        cout<<"Account Number="<<accno<<endl;
        cout<<"Balance="<<balance<<endl;
        cout<<"Interest Rate="<<interestrate<<"%"<<endl;
    }
};
class CheckingAccount
{
    public:
    string accname;
    int accno;
    double balance;
    double transactionfee;
    CheckingAccount(int no, string name, double bal, double fee)
    {
        accno=no;
        accname=name;
        balance=bal;
        transactionfee=fee;
    }
    void deposit(double amount)
    {
        if(amount>0)
        {
            balance+=amount;
            cout<<"Deposited="<<amount<<endl;
        }
    }
    void withdraw(double amount)
    {
        double total=amount+transactionfee;
        if(total<=balance)
        {
            balance-=total;
            cout<<"Withdrawn="<<amount<<"("<<transactionfee<<"Fee applied"<<endl;
        }
        else
        {
            cout<<"Insufficient balance for withdrawn and fee!"<<endl;
        }
    }
    void display()
    {
        cout<<"\n[CHECKING ACCOUNT]"<<endl;
        cout<<"Account Holder="<<accname<<endl;
        cout<<"Account Number="<<accno<<endl;
        cout<<"Balance="<<balance<<endl;
        cout<<"Transaction Fee="<<transactionfee<<endl;
    }
};
int main()
{
    SavingAccount saving(101,"Shreya Dhone",5000.0,3.0);
    saving.display();
    saving.deposit(1000);
    saving.withdraw(2000);
    saving.applyInterest();
    saving.display();
    CheckingAccount checking(102,"Anushka Salunkhe",3000.0,20.0);
    checking.display();
    checking.deposit(1500);
    checking.withdraw(2000);
    checking.display();
    return 0;
}
