#include<iostream>
using namespace std;
class Distance
{
    public:
    int feet,inch;
    Distance(int f,int i)
    {
        feet=f;
        inch=i;
    }
    void operator-()
    {
        feet--;
        inch--;
        cout<<"\nFeet and Inch (Decrement):"<<feet<<"'"<<inch;
    }
};
int main()
{
    Distance d1(8,9);
    -d1;
    return 0;
}

Output:
Feet and Inch (Decrement):7'8
