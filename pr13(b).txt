
#include <iostream>
using namespace std;


class MyException : public exception
{
public:
    const char* what() const throw()
    {
        return "Attempted to divide by zero!\n";
    }
};

int main()
{
    try {
        int x, y;
        cout << "Enter two numbers: ";
        cin >> x >> y;
        if (y == 0) {
            MyException z;
            throw z;
        } else {
            cout << x / y << endl;
        }
    }
    catch (exception &e) {   // <-- fixed
        cout << e.what();
    }
}


Output:
Enter two numbers: 5 0
Attempted to divide by zero!
